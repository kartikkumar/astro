100template <typename Real, typename Vector3>

102 const Real radiationPressureCoefficient,
103 const Vector3& unitVectorToSource,
104 const Real radius,
105 const Real bulkDensity)
106{
107 Vector3 acceleration = unitVectorToSource;
108
109 const Real preMultiplier = -radiationPressure

153template <typename Real, typename Vector3>
154Vector3 computeCannonballPoyntingRobertsonDragAcceleration(
155 const Real radiationPressure,
156 const Real radiationPressureCoefficient,
157 const Vector3& unitVectorToSource,
158 const Real radius,
159 const Real bulkDensity,
160 const Vector3& velocity)
161{
162 Vector3 acceleration = unitVectorToSource;
163
164 const Real preMultiplier = radiationPressure
165 * radiationPressureCoefficient
166 * 0.75 / (radius * bulkDensity * ASTRO_SPEED_OF_LIGHT);
167
168 acceleration[0]
169 = preMultiplier * (unitVectorToSource[0] * unitVectorToSource[0] + 1.0) * velocity[0];
170 acceleration[1]
171 = preMultiplier * (unitVectorToSource[1] * unitVectorToSource[1] + 1.0) * velocity[1];
172 acceleration[2]
173 = preMultiplier * (unitVectorToSource[2] * unitVectorToSource[2] + 1.0) * velocity[2];
174
175 return acceleration;
176}
